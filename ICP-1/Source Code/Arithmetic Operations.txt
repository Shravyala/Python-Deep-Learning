# Performing arithmetic operations on two numbers

num1String=input('Enter first number:')

num2String=input('Enter second number:')

num1=int(num1String)

num2=int(num2String)

# Addition of two numbers

Sum=int(num1)+int(num2)
print("the sum of", num1, "and", num2, "is", Sum)

# Subtraction of two numbers

Sub=int(num1)-int(num2)
print("the difference of", num1, "and", num2, "is", Sub)

# Multiplication of two numbers

Mul=int(num1)*int(num2)
print("the product of", num1, "and", num2, "is", Mul)

# Division of two numbers

Div=int(num1)/int(num2)
print("the division of", num1, "and", num2, "is", Div)
